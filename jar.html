    <script>

        // --- YOUR DATA ARRAY ---

        const myMessages = [

            "I love how your eyes crinkle when you laugh. 🥹",

            "I'm sorry if I'm ever unavailable, you're always my priority.",

            "You are the best part of my 'forever lore'.",

            "Remember: I am always trying my best for us. ❤️",

            "You make even the most boring days feel like an adventure.",

            "I love you more than words can actually put into code. 🥰",

            "You're my favorite person to do absolutely nothing with."

        ];


        const display = document.getElementById('message-display');

        const text = document.getElementById('note-text');


        function pullMessage() {

            // Logic: Random Index (just like Random.Range in C#)

            const randomIndex = Math.floor(Math.random() * myMessages.length);

            text.innerText = myMessages[randomIndex];

            display.classList.add('show');

        }


        function closeNote(event) {

            event.stopPropagation(); // Prevents re-triggering the jar click

            display.classList.remove('show');

        }
