    <script>

        const IMAGE_URL = 'https://i.postimg.cc/LhDPzXGp/Shurya-3.jpg'; 

        const SUCCESS_MESSAGE = "Congratulations you won a lifetime worth of apologies and endless \" I love you's ❤️\"..I'm sorry babu..I'm sorry if sometimes I'm too much or too less available for you..but i swear i try my best to be the best verisons of ourselves and not to let our \"us\" fade away..I LOVE YOUU SOO MUCH MY BABYYY!! 🥹";


        const container = document.getElementById('puzzle-container');

        const overlay = document.getElementById('overlay');

        const noBtn = document.getElementById('no-btn');

        let tiles = [];

        let selectedTile = null;

        let noBtnScale = 1;


        // Floating Hearts

        function createHeart() {

            const heart = document.createElement('div');

            heart.className = 'floating-heart';

            heart.innerHTML = ['❤️', '🌸', '✨', '💖'][Math.floor(Math.random()*4)];

            heart.style.left = Math.random() * 100 + 'vw';

            heart.style.animationDuration = Math.random() * 3 + 4 + 's';

            heart.style.fontSize = Math.random() * 1 + 0.8 + 'rem';

            heartContainer.appendChild(heart);

            setTimeout(() => heart.remove(), 7000);

        }

        const heartContainer = document.getElementById('heart-container');

        setInterval(createHeart, 400);


        // Puzzle Logic

        function initPuzzle() {

            let positions = Array.from({length: 9}, (_, i) => i);

            let shuffled = [...positions].sort(() => Math.random() - 0.5);

            shuffled.forEach((pos, index) => {

                const tile = document.createElement('div');

                tile.className = 'tile';

                tile.dataset.correctPos = pos;

                tile.style.order = index;

                const x = (pos % 3) * 50;

                const y = Math.floor(pos / 3) * 50;

                tile.style.backgroundImage = `url(${IMAGE_URL})`;

                tile.style.backgroundPosition = `${x}% ${y}%`;

                tile.addEventListener('click', () => handleTileClick(tile));

                container.appendChild(tile);

                tiles.push(tile);

            });

        }


        function handleTileClick(tile) {

            if (!selectedTile) {

                selectedTile = tile;

                tile.classList.add('selected');

            } else {

                if (selectedTile === tile) {

                    selectedTile.classList.remove('selected');

                    selectedTile = null;

                    return;

                }

                const tempOrder = selectedTile.style.order;

                selectedTile.style.order = tile.style.order;

                tile.style.order = tempOrder;

                selectedTile.classList.remove('selected');

                selectedTile = null;

                setTimeout(checkWin, 400);

            }

        }


        function checkWin() {

            const allCorrect = tiles.every(tile => parseInt(tile.style.order) === parseInt(tile.dataset.correctPos));

            if (allCorrect) {

                document.getElementById('final-message').innerText = SUCCESS_MESSAGE;

                document.getElementById('message-modal').style.display = 'block';

                overlay.style.display = 'block';

            }

        }


        function openConsent() {

            document.getElementById('message-modal').style.display = 'none';

            document.getElementById('consent-modal').style.display = 'block';

        }


        function showFinal() {

            document.getElementById('consent-modal').style.display = 'none';

            document.getElementById('final-modal').style.display = 'block';

        }


        function moveNoButton() {

            const modal = document.getElementById('consent-modal');

            const rect = modal.getBoundingClientRect();

            const x = Math.random() * (rect.width - 100) - (rect.width/2) + 50;

            const y = Math.random() * (rect.height - 100) - (rect.height/2) + 50;

            noBtnScale -= 0.15;

            if (noBtnScale < 0.2) noBtnScale = 0.2;

            noBtn.style.transform = `translate(${x}px, ${y}px) scale(${noBtnScale})`;

        }


        noBtn.addEventListener('mouseover', moveNoButton);

        noBtn.addEventListener('touchstart', (e) => { e.preventDefault(); moveNoButton(); });


        initPuzzle();
