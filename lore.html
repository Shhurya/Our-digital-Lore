    <script>

        // --- ADD YOUR DATA HERE ---

        const scrapbookData = [

            {

                image: "https://i.postimg.cc/LhDPzXGp/Shurya-3.jpg",

                date: "OCT 12, 2025",

                text: "The very first glimpse of us. I'm so glad this happened."

            },

            {

                image: "https://via.placeholder.com/300x400",

                date: "NOV 05, 2025",

                text: "Second memory. You look so beautiful here."

            },

            {

                image: "https://via.placeholder.com/300x400",

                date: "DEC 20, 2025",

                text: "Third memory. Building our lore, one day at a time."

            }

        ];


        const book = document.getElementById('book');

        let flippedCount = 0;


        function buildBook() {

            // We loop forward. The first item in array should be the first page under the cover.

            scrapbookData.forEach((data, index) => {

                const pageLeaf = document.createElement('div');

                pageLeaf.className = 'page';

                // Decrease Z-index so pages stack correctly (Cover is 100)

                pageLeaf.style.zIndex = 99 - index;


                pageLeaf.innerHTML = `

                    <div class="page-front">

                        <div class="polaroid">

                            <img src="${data.image}">

                            <div class="date-tag">${data.date}</div>

                        </div>

                        <p class="description">${data.text}</p>

                    </div>

                    <div class="page-back">

                        <p class="description">Keep going... 🥰</p>

                    </div>

                `;

                book.appendChild(pageLeaf);

            });

        }


        book.addEventListener('click', () => {

            const allLeaves = document.querySelectorAll('.page');

            

            if (flippedCount < allLeaves.length) {

                allLeaves[flippedCount].classList.add('flipped');

                flippedCount++;


                if (flippedCount === 1) {

                    book.style.transform = "translateX(160px)";

                }

            } else {

                // Reset

                allLeaves.forEach(leaf => leaf.classList.remove('flipped'));

                book.style.transform = "translateX(0)";

                flippedCount = 0;

            }

        });


        buildBook();

    </script>
